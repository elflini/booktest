load("claim.RData")
library(tidyverse)

R은 사용자가 직접 만들어 사용하는 사용자 정의 함수를 쉽게 생성 할 수 있으며, 정의 방법은 다음과 같다.

함수이름 <- function(인수) {
실행문
(출력문)
}

사용자 정의 함수를 만들때는 반드시 제일 마지막 구문이 출력을 행하는 구문을 지정해야 한다.

간단하게 평균을 구하는 함수를 생성해보면 다음과 같다.

meanx <- function(x) {
  sumx <- sum(x)
  meanx <- sumx/length(x)
  meanx
}

meanx(1:10)
## [1] 5.5

위의 프로그램을 살펴보면 meanx 라는 평균을 구하는 함수를 만들었다. 이때 meanx() 함수의 인수는 ’x’이며 이 인수는 함수내의 sum(x)에 사용된 ’x’로 바로 매칭된다. 즉, 예를들어 meanx(1:10)으로 두게 되면 ’1:10’을 인수 ’x’가 받게되고, 이 인수값은 함수내의 sum(x)에서 x의 값으로 들어가게 된다. 그리고 함수의 마지막 구문은 meanx를 출력하라는 구문으로 지정되었다. 함수 사용법은 ’함수이름(인수)’를 사용하면 되고, 따라서 meanx(1:10)은 1부터 10까지의 숫자의 평균을 구하는 함수로 작동한다.

다음의 예제를 살펴보자. 만약 t20에서 주상병 또는 제1부상병의 값이 인수로 지정한 값이면 1, 아니면 0의 값을 갖고 이러한 변수 이름을 disease로 지정할 수있게 하는 함수를 생각해보자.

diseasex <- function(x, icd) {
  require(tidyverse)
  
  
  x1 <- x
  x1 <- x1 %>% 
    mutate(disease=ifelse(substr(sick_sym1,1,3) %in% icd | substr(sick_sym2,1,3) %in% icd ,1,0))
  
  x1 <- x1[c("rn_indi","rn_key","disease")]
  x1
}
